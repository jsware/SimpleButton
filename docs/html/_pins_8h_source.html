128 /**
129 * Debounce the pin readings to get a stable state of the pin. In order to
130 * save memory, virtual functions are *not* used (which can consume almost
131 * 1K of memory to deal with). Runtime polymorphism is not required, so
132 * avoiding virtual functions saves memory.
133 */
134 inline void readPin() {
135 byte currReading = digitalRead(pin_);
136
137 // New reading, so start the debounce timer.
138 if (currReading != prevReading_) {
139 debounceStart_ = millis();
140 } else if ((millis() - debounceStart_) >= debounceTime_) {
141 // Use reading if we have the same reading for >= DELAY ms.
142 currState_ = currReading;
143 }
144
145 prevReading_ = currReading;
146 }
147
148 public:

150 // Accessors...
151 //
152
153 /**

155 *
156 * @returns The number of milliseconds of debounce time.
157 */
158 inline uint8_t debounceTime() const {
159 return debounceTime_;
160 }

207 * state changes, remember the previous state so clicks (press then

209 * for a change in debounced state, remembering the previous state.
210 */
211 inline void readPin() {
212 uint8_t currState = currState_; // Remember current state.
213
214 DebouncedPin::readPin();
215
216 // Save previous state if it changed.
217 if(currState != currState_) {
218 prevState_ = currState;
219 }
220 }
