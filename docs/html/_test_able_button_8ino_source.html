2 * @file TestAbleButton.ino Test the Arduino Button Library Extension (ABLE).
3 * This program will periodically output "Looping..." to the Serial port. As you
4 * press buttons connected to pins 2 and 3 it will assert the button settings
5 * in the callback fuctions and in the main loop.

38 * Control TestButton. Called repeatedly in a loop.

40void loop() {
41 static uint16_t count = 0; ///< Counter to preiodically print OK message.
42 if(!count++) Serial.println(F("Looking OK..."));
43
44 btnList.handle(); // Handle features of the buttons (debouncing, clicks & callbacks).
45
46 if(btnList.anyClicked()) {
47 led = !led;
48 }
49
50 if(anyReleased) {
51 assert(btnList.allClicked() == false); // Not all buttons should be clicked.
52 assert(btnList.anyClicked() == true); // Clicked will be set after button released.
53 assert(btnList.resetClicked() == true); // Clear any clicked status.
54 assert(btnList.anyClicked() == false); // No buttons should be clicked now.
55 anyReleased = false; // Handled most recent release callback signal.
56 } else {
57 assert(btnList.allPressed() == false); // Not all buttons normally pressed.
58 assert(btnList.allClicked() == false); // Not all buttons normally clicked.
59 assert(btnList.anyClicked() == false); // Only clicked when anyReleased set by released callback.
60 }
61
62 digitalWrite(LED_BUILTIN, led);
63}
