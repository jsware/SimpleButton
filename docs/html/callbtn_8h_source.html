41 /**
42 * Handle the button. Called from loop() of an Arduino program.
43 */
44 void handle() {
45 if(!pressed_ && BaseButton::isPressed()) {
46 pressed_ = true;
47 if(onPressed_) {
48 onPressed_(this);
49 }
50 } else if(pressed_ && !BaseButton::isPressed()) {
51 pressed_ = false;
52 if(onReleased_) {
53 onReleased_(this);
54 }
55 }
56 }
