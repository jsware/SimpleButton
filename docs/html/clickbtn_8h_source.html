14 /**
15 * Clickable button template for clickable buttons.
16 *
17 * @param ButtonType Either a PullupButtonType or PulldownButtonType
18 * matching the resistor configuration of the button. Will
19 * default to PullupButtonType as the simplest button
20 * connection.
21 */
22 template <typename ButtonType = class PullupButtonType, typename BaseButton = TBasicButton<ButtonType> >
23 class TClickableButton: public BaseButton {
24 public:
25 // ClickableButton must be connected via a pin.
26 TClickableButton() = delete;
27
28 /**
29 * Create a clickable button on the specified pin.
30 *
31 * @param pin The pin connected to the button.
32 * @param onClicked The function to call when the button is clicked.
33 * @param id Callback identifier for the button (default auto-assigned).
34 */
35 TClickableButton(uint8_t pin,
36 void(*onClicked)(uint8_t) = 0,
37 uint8_t id = autoId())
38 :BaseButton(pin), id_(id), state_(ButtonState::NONE), onClicked_(onClicked) {}
39
40 /**
41 * Handle the button. Called from loop() of an Arduino program.
42 */
43 void handle() {
44 if(!(state_ == ButtonState::PRESSED) && BaseButton::isPressed()) {
45 state_ = ButtonState::PRESSED;
46 } else if(state_ == ButtonState::PRESSED && !BaseButton::isPressed()) {
47 if(onClicked_) {
48 state_ = ButtonState::NONE;
49 onClicked_(id_);
50 } else {
51 state_ = ButtonState::CLICKED;
52 }
53 }
54 }
55
56 /**
57 * Set a (new) on clicked callback function.
58 *
59 * @param callbackFn The function to call when the button is clicked. Use
60 * 0 to clear the on-clicked callback.
61 */
62 void onClicked(void(*callbackFn)(uint8_t)) {
63 onClicked_ = callbackFn;
64 }
65
66 /**
67 * Determine if button was clicked and reset the clicked state.
68 *
69 * @return True if button was clicked, false if not.
70 */
71 bool wasClicked() {
72 bool rc = (state_ == ButtonState::CLICKED);
73 state_ = ButtonState::NONE;
74 return rc;
75 }
76
77 private:
78 uint8_t id_; ///< Identifier for the button passed to callback functions.
79 uint8_t state_; ///< Indicates the state of the button.
80 void (*onClicked_)(uint8_t); ///< Callback function for button click.
81 };
82}
