13 * and pull-up resistor circuits specified using the Circuit.
14 *
15 * @param Circuit Either a PullupResistorCircuit or PulldownResistorCircuit

108 * state, otherwise the compile will fail with undefined reference to the

110 *
111 * See the Button<PulldownResistorCircuit, ClickerPin>::isClicked() and

113 *
114 * @return True if clicked else false.

116 bool isClicked() const;

123 /**
124 * Specialisation of Button::isClicked() for clickable pins using pulldown
125 * resistor circuits.
126 *

129 template <>
130 inline bool Button<PulldownResistorCircuit, ClickerPin>::isClicked() const {
131 return currState_ == PulldownResistorCircuit::BUTTON_RELEASED && prevState_ == PulldownResistorCircuit::BUTTON_PRESSED;
132 }
133

143 }
